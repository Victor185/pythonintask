import random
sputniki=("Деймос","Фобос")
sputn=random.randint(0,1)
rand=sputniki[sputn]
print("Я загадал название одного из спутников марса")
#print(rand)
otvet=0 
while(otvet)!=(rand):
    otvet=input("Введите назмание спутника:")
    if(otvet)!=(rand):
        print("Вы не угадали,попробуйте снова.")
    elif(otvet)==(rand):
        print("Вы угадали.")
        break
input("Нажмите Enter для выхода")
